% ==== 1. Motivation: the policy question, no exhibit ====
\begin{frame}{Digital infrastructure could close the small-firm productivity gap}
  \begin{itemize}
    \item Small firms employ most workers but modernize last
          \graycite{(Syverson '11)}   % Intro, p.1
    \item Fast internet could raise their productivity at low fiscal cost
    \item[$\Rightarrow$] Yet no causal estimate exists for the firms that need it most
  \end{itemize}
\end{frame}

% ==== 2. Question: stated as economics, antecedents as gray citations ====
\begin{frame}{Existing work has bypassed the small-firm segment}
  \begin{itemize}
    \item Broadband's effect is measured in aggregate, not for small firms
    \item Adoption is endogenous to a firm's own prospects
    \item[$\Rightarrow$] We exploit access timing that firms did not choose
  \end{itemize}
  \medskip
  \graycite{Prior work: Czernich et al.\ '13; Akerman et al.\ '15; Hjort--Poulsen '19}  % Intro/Lit, p.2, p.4
\end{frame}

% ==== 3. This paper: THE PUNCHLINE SLIDE (lands by slide 4) ====
\begin{frame}{This paper}
  \begin{enumerate}
    \item First causal small-firm estimate: productivity rises \KeyIdea{3.4\%} (0.034)   % Table 2 col 1, p.7
    \item Gains rise with communication intensity, as a simple model predicts   % Table 4, p.9; Prop 1, p.3
    \item Back-of-the-envelope: the program repaid its cost within six years   % Sec 7, p.8-9
  \end{enumerate}
  \medskip
  \textbf{Implication:} universal broadband is among the cheapest productivity
  policies for a small open economy.
\end{frame}

% ==== 8. Policy magnitude: the third contribution ====
\begin{frame}{The program paid for itself within six years}
  \begin{ResultBox}[Policy return]
    A 3.4\% productivity gain for the average connected small firm is worth about
    \KeyIdea{410 million euros a year}. Against the fund's 2.3 billion euro
    construction cost, the small-firm channel alone repays the public investment
    within six years.
  \end{ResultBox}   % Sec 7, p.8-9
  \medskip
  \RunIn{Scope:} the mechanism is communication cost, not a Nordavian
  particularity, so the logic travels to other broadband programs.   % Conclusion, p.9
\end{frame}

% ==== 9. Conclusion: mirrors "This paper" ====
\begin{frame}{Conclusion}
  \begin{enumerate}
    \item Broadband arrival raises small-firm labor productivity by \KeyIdea{3.4\%}   % Table 2 col 1, p.7
    \item Gains concentrate where firms rely on communication-intensive tasks
    \item The program repaid its cost within six years
  \end{enumerate}
  \medskip
  \textbf{Implication:} universal broadband is among the cheapest productivity
  policies for a small open economy.
\end{frame}

% A2 — Model and estimating equation
\begin{frame}{Model and estimating equation}
  \hypertarget{app:model}{}
  A task-outsourcing model delivers the estimating equation   % Sec 2.2, eq (5), p.3
  \[
    y_{it}-\ell_{it} = \alpha_i + \lambda_t
      + \underbrace{\beta\, B_{mt}}_{\textcolor{cAccentA}{\text{average effect}}}
      + \underbrace{\gamma\,(B_{mt}\times\theta_i)}_{\textcolor{cAccentC}{\text{complementarity}}}
      + \varepsilon_{it},
  \]
  with $B_{mt}$ a connected-municipality flag and $\theta_i$ communication intensity.
  \medskip
  \begin{ResultBox}[Proposition 1]
    The gain is strictly increasing in communication intensity: $G'(\theta_i) > 0$ on $[0,1]$.   % Prop 1, p.3
  \end{ResultBox}
  \BackButton{main:mech}
\end{frame}

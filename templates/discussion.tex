% See references/discussant.md for the genre rules.
\documentclass[11pt, aspectratio=169]{beamer}
\usepackage{econ-slides-boxed}   % house/clean also work; boxed suits policy rooms

\title{[The paper's title]}
\author{by [Paper's authors]\\[1.2em]
        Discussion by [Discussant] ([Affiliation])}
\date{[Conference], [Month Year]}

\begin{document}

\begin{frame}[plain]
  \titlepage
  % \vfill\centering\scriptsize\textit{Disclaimer: views are my own.}
\end{frame}
\addtocounter{framenumber}{-1}

% ---- One-slide summary (compress or drop if the talk was clear) ----
\begin{frame}[t]{The paper in one slide}
  \medskip
  \begin{itemize}
    \item[] \RunIn{Question:} [one line]
    \item[] \RunIn{Design:} [one line: data + identification/model]
    \item[] \RunIn{Findings:} [one line with the headline number]  % p._, Table _
  \end{itemize}
\end{frame}

% ---- What works ----
\begin{frame}[t]{What the paper does well}
  \medskip
  \begin{itemize}
    \item {[Genuine, specific strength 1]}
    \item {[Strength 2 — e.g. what the data uniquely allow]}
    \item {[Where it advances the literature, one line]}
  \end{itemize}
  \medskip
  My comments focus on [the theme connecting the comments].
\end{frame}

% ---- Comment slides: the title IS the comment ----
\begin{frame}[t]{[Comment 1 as a one-line actionable assertion or question]}
  \medskip
  \begin{itemize}
    \item \RunIn{The paper:} [what it currently does]  % Section _, p._
    \item {[The concern, as analysis — own calculations labeled as yours]}
    \item[$\Rightarrow$] [Concrete suggestion the authors can act on]
  \end{itemize}
\end{frame}

\begin{frame}[t]{[Comment 2]}
  \medskip
  \begin{itemize}
    \item \RunIn{The paper:} [...]
    \item {[...]}
    \item[$\Rightarrow$] [...]
  \end{itemize}
\end{frame}

\begin{frame}[t]{[Comment 3 — optional; two strong comments beat three thin ones]}
  \medskip
  \begin{itemize}
    \item \RunIn{The paper:} [...]
    \item {[...]}
    \item[$\Rightarrow$] [...]
  \end{itemize}
\end{frame}

% ---- Summary: fair restatement + comments as one-liners; end positive ----
\begin{frame}[t]{Summary}
  \medskip
  \begin{itemize}
    \item {[The paper's takeaway, restated fairly]}
    \item[$\Rightarrow$] [Comment 1 in one line]
    \item[$\Rightarrow$] [Comment 2 in one line]
  \end{itemize}
  \medskip
  [One warm closing line: why this paper is worth improving.]
\end{frame}

% ---- Optional overflow, usually not presented ----
\begin{frame}[noframenumbering]{Other comments}
  \begin{itemize}
    \item {[Smaller point 1 — minor specification/presentation suggestions]}
    \item {[Smaller point 2. Typos go to the authors privately, not here.]}
  \end{itemize}
\end{frame}

\end{document}

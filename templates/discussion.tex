\begin{frame}{The paper in one slide}
  \begin{itemize}\setlength{\itemsep}{0.7em}
    \item[] \RunIn{Question:} [The authors' question, stated in their favor]
      \begin{itemize}\item {[Why the answer matters]}\end{itemize}
    \item[] \RunIn{Design:} [Data plus identification or model]
      \begin{itemize}\item {[The comparison or discipline doing the work]}\end{itemize}
    \item[] \RunIn{Finding:} [Headline pattern and magnitude]  % p._, Table _
      \begin{itemize}\item {[Authors' preferred interpretation]}\end{itemize}
  \end{itemize}
  \bigskip
  \textbf{If the authors are right:} [What readers should believe or do differently.]
\end{frame}

\begin{frame}{What the paper does well}
  \begin{itemize}\setlength{\itemsep}{0.7em}
    \item \textbf{[Specific strength in the question or setting]}
      \begin{itemize}\item {[Why it changes what we can learn]}\end{itemize}
    \item \textbf{[Specific strength in data, model, or design]}
      \begin{itemize}\item {[What the paper can show that prior work cannot]}\end{itemize}
    \item \textbf{[Specific strength in execution]}
      \begin{itemize}\item {[The exhibit or analysis that earns this claim]}\end{itemize}
  \end{itemize}
  \bigskip
  \textbf{My comments share one goal:} [One theme connecting them.]
  \par\medskip
  \textbf{Opportunity:} [An improvement that preserves the strongest contribution.]
\end{frame}

% Duplicate this anatomy for a third comment only when it is equally strong.
\begin{frame}[t]{[State the first actionable comment]}
  \medskip
  \begin{itemize}
    \item \RunIn{The paper:} [What it currently assumes or shows]  % Section _, p._
      \begin{itemize}\item {[The authors' logic, stated fairly]}\end{itemize}
    \item \RunIn{Concern:} [Why the present evidence leaves an ambiguity]
      \begin{itemize}\item {[One concrete implication, example, or own calculation]}\end{itemize}
    \item \RunIn{Evidence needed:} [The observation that separates the two readings]
  \end{itemize}
  \begin{ResultBox}[Concrete revision]
    [One feasible analysis, rewrite, or exhibit the authors can add.]
  \end{ResultBox}
  \smallskip
  \textbf{Payoff:} [What we could conclude after that revision.]
\end{frame}

\begin{frame}[t]{[State the second actionable comment]}
  \medskip
  \begin{itemize}
    \item \RunIn{The paper:} [What it currently assumes or shows]
      \begin{itemize}\item {[The strongest case for the current choice]}\end{itemize}
    \item \RunIn{Concern:} [The unresolved interpretation or design issue]
      \begin{itemize}\item {[Why it matters for the paper's main claim]}\end{itemize}
    \item \RunIn{Evidence needed:} [The falsification or comparison that would decide]
  \end{itemize}
  \begin{ResultBox}[Concrete revision]
    [The smallest change that would separate the competing explanations.]
  \end{ResultBox}
  \smallskip
  \textbf{Payoff:} [How the change sharpens the paper's main contribution.]
\end{frame}

\begin{frame}[t]{Summary}
  \medskip
  \begin{itemize}\setlength{\itemsep}{0.8em}
    \item \textbf{[Restate the paper's contribution generously.]}
      \begin{itemize}\item {[The evidence that makes it worth reading]}\end{itemize}
    \item \RunIn{First:} [First actionable comment and its concrete revision]
      \begin{itemize}\item {[What the revision would clarify]}\end{itemize}
    \item \RunIn{Second:} [Second actionable comment and its concrete revision]
      \begin{itemize}\item {[What the revision would unlock]}\end{itemize}
  \end{itemize}
  \smallskip
  [One warm closing line about the paper's promise.]
  \par\medskip
  \textbf{Revision path:} [Resolve the first comment before the second.]
\end{frame}

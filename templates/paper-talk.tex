% ---- 1. Key question: motivation and question in one audience-facing frame. ----
% A source-native motivation figure may precede this when the audience needs
% it. A concise question belongs in the title. If it cannot be made concise,
% use {Key question} and place \KeyQuestion in the body instead. Give the
% answer as soon as it is intelligible; do not force a fixed slot.
\begin{frame}[t]{\KeyQuestion}
  \medskip
  \RunIn{Why this matters:}
  \begin{itemize}\setlength{\itemsep}{0.55em}
    \item {[One source-supported fact establishing scale or importance]}
    \item {[How this margin changes behavior, welfare, or policy]}
  \end{itemize}

  \medskip
  \RunIn{What makes it hard:}
  \begin{itemize}\setlength{\itemsep}{0.55em}
    \item {[The central selection, equilibrium, or measurement problem]}
    \item {[The comparison a convincing answer must defend]}
  \end{itemize}
  \medskip
  \graycite{[Two to four verified closest antecedents, inline.]}
\end{frame}
